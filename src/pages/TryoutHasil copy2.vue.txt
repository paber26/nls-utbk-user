<template>
  <body class="bg-bgsoft font-poppins">
    <div class="flex min-h-screen">
      <Sidebar></Sidebar>

      <!-- ================= MAIN ================= -->
      <main class="flex-1">
        <!-- Topbar -->
        <header class="bg-white border-b px-4 md:px-6 py-4">
          <h1 class="text-lg font-semibold text-slate-800">Hasil Tryout</h1>
          <p class="text-sm text-slate-500">Ringkasan hasil pengerjaan tryout</p>
        </header>

        <!-- Content -->
        <div class="px-4 md:px-6 py-6 w-full">
          <!-- Loading State -->
          <div v-if="loading" class="flex flex-col items-center justify-center py-20">
            <div class="w-12 h-12 border-4 border-indigo-200 border-t-indigo-600 rounded-full animate-spin"></div>
            <p class="mt-4 text-slate-500">Memuat hasil tryout...</p>
          </div>

          <!-- Content After Loaded -->
          <div v-else>
            <section class="bg-white rounded-xl border p-10 text-center">
              <div class="flex flex-col items-center">
                <div class="w-16 h-16 flex items-center justify-center rounded-full bg-emerald-100 mb-4">
                  <svg
                    xmlns="http://www.w3.org/2000/svg"
                    class="w-8 h-8 text-emerald-600"
                    fill="none"
                    viewBox="0 0 24 24"
                    stroke="currentColor"
                  >
                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M5 13l4 4L19 7" />
                  </svg>
                </div>

                <h2 class="text-2xl font-bold text-slate-800 mb-3">Terima Kasih Sudah Mengikuti Tryout JSS 🎉</h2>

                <p class="text-slate-600 max-w-xl">
                  Jawaban Anda telah berhasil dikumpulkan. Nilai dan peringkat akan diumumkan serta diinformasikan pada
                  waktu yang telah ditentukan oleh panitia.
                </p>

                <RouterLink
                  to="/tryout"
                  class="mt-6 inline-block px-6 py-3 rounded-lg bg-primary text-white hover:bg-indigo-700"
                >
                  Kembali ke Halaman Tryout
                </RouterLink>
              </div>
            </section>
          </div>
        </div>
      </main>
    </div>
  </body>
</template>

<script setup>
import { RouterLink, RouterView } from "vue-router"

import Sidebar from "../components/layout/Sidebar.vue"

import { onMounted, ref } from "vue"
import { useRoute } from "vue-router"
import api from "../services/api"

const route = useRoute()
const id = route.params.id
// console.log("ini attempt id", attemptId)

const hasil = ref(null)
const ringkasan = ref({
  benar: 0,
  salah: 0,
  kosong: 0
})
const navigasi = ref([])

const loading = ref(true)

const loadHasil = async () => {
  loading.value = true
  try {
    const res = await api.get(`/user/tryout/hasil/${id}`)
    console.log("Hasil tryout data:", res)
    hasil.value = res.data

    ringkasan.value.nilai_poin = res.data.nilai_poin
    ringkasan.value.benar = res.data.benar
    ringkasan.value.salah = res.data.salah
    ringkasan.value.kosong = res.data.kosong
    navigasi.value = res.data.navigasi
  } catch (e) {
    console.error("Gagal memuat hasil tryout", e)
  } finally {
    loading.value = false
  }
}

onMounted(loadHasil)
</script>
